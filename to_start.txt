1. docker compose up -d
    - Check docker ps
2. kafka-topics --create --bootstrap-server localhost:9092 --replication-factor 1 --partitions 2 --topic market-events
    - if you want to remove each message per key 
        - kafka-topics --create --bootstrap-server localhost:9092 --replication-factor 1 --partitions 2 --topic market-events --config cleanup.policy=compact
    - Check kafka-topics --list --bootstrap-server localhost:9092
3. activate pyton 
4. start consumers 
    - python consumer/consumer_stats.py (python -m consumer.consumer_stats)
    - python consumer/consumer_dashboard_csv.py (python -m consumer.consumer_dashboard_csv)
5. start producer 
    - python producer/producer.py (python -m producer.producer)

If needed to delete topic:
    - kafka-topics --delete --bootstrap-server localhost:9092 --topic market-events

See all message by key kafka-console-consumer --bootstrap-server localhost:9092 --topic market-events --from-beginning \
  --property print.key=true \
  --property print.value=true \
  --property key.separator=" | "

To shut down docker 
    - docker kill <ID>
